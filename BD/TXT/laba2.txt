ALTER SESSION SET "_ORACLE_SCRIPT" = TRUE;
CREATE TABLESPACE TS_KND
    DATAFILE 'TS_KND.dat'
    SIZE 7M
    REUSE
    AUTOEXTEND ON NEXT 5M MAXSIZE 30M;

CREATE TEMPORARY TABLESPACE TS_KND_TEMP
    TEMPFILE 'TS_KND_TEMP'
    SIZE 5M
    REUSE
    AUTOEXTEND ON NEXT 3M MAXSIZE 20M;
alter session set "_ORACLE_SCRIPT"=true;

CREATE ROLE RL_KNDCORE;

GRANT CONNECT,
    CREATE TABLE,
    CREATE VIEW,
    CREATE ANY PROCEDURE TO RL_KNDCORE;

CREATE PROFILE PF_KNDCORE LIMIT
    PASSWORD_LIFE_TIME 180
    SESSIONS_PER_USER 3
    FAILED_LOGIN_ATTEMPTS 7
    PASSWORD_LOCK_TIME 1
    PASSWORD_REUSE_TIME 10
    PASSWORD_GRACE_TIME DEFAULT
    CONNECT_TIME 180
    IDLE_TIME 30;

CREATE USER KNDCORE
    IDENTIFIED BY KNDCORE12345
    DEFAULT TABLESPACE TS_KND
    TEMPORARY TABLESPACE TS_KND_TEMP
    PROFILE PF_KNDCORE
    ACCOUNT UNLOCK
    PASSWORD EXPIRE;
grant create session to KNDCORE;
SELECT TABLESPACE_NAME
FROM dba_tablespaces;

SELECT tablespace_name, file_name
FROM dba_data_files;

SELECT *
FROM DBA_ROLES
WHERE ROLE = 'RL_KNDCORE';

SELECT role
FROM dba_roles
WHERE role = 'RL_KNDCORE';

SELECT privilege
FROM dba_sys_privs;

SELECT profile
FROM dba_profiles;

SELECT profile, resource_name, limit
FROM dba_profiles
WHERE profile = 'PF_KNDCORE';

SELECT profile, resource_name, limit
FROM dba_profiles
WHERE profile = 'DEFAULT';
